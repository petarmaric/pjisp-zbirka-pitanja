Za identifikaciju pojedine pozicije u vektoru koristi se:
+ indeks
- pokazivac
- genericki pokazivac
- viseci pokazivac
.

Dvodimenzionalna generalizacija vektora se zove:
+ matrica
- binarno stablo
- STEK
- DEK
.

Ako je zadat dvostruki pokazivac p:
    int **p;
    p = calloc(10,sizeof(int*));
onda p[i][j] znaci isto sto i:
+*(*(p+i)+j)
-**(p+i+j)
-**p+i+j
-*p+(i+j)
-**p+(i+j)
.

Sta ispisuje sledeci program:
	#include <stdio.h>
	#include <stdlib.h>
	int main(){
		int **a, n, i, j;

		printf("n = ");
		scanf("%d", &n);

		a = calloc(n, sizeof(int*));
		for (i=0; i<n; i++) {
			*(a+i) = calloc(n, sizeof(int));
			for (j=0; j<n; j++)
				*(*(a+i)+j) = rand()/((double)RAND_MAX +1) * 10;
		}

		for (i=0; i<n; printf("\n"), i++)
			for (j=0; j<n; j++)
				printf("%d ", * (*(a+i)+j));

		return 0;
	}
-sadrzaj matrice vrstu po vrstu, a potom sporednu dijagonalu, sleva-udesno
-sadrzaj matrice vrstu po vrstu, a potom glavnu dijagonalu, sleva-udesno
+sadrzaj matrice vrstu po vrstu
-sporednu dijagonalu sdesna-ulevo
-glavnu dijagonalu sdesna-ulevo
-sporednu dijagonalu sleva-udesno

Sta ispisuje sledeci program:
	#include <stdio.h>
	#include <stdlib.h>
	int main(){
		int **a, n, i, j;

		printf("n = ");
		scanf("%d", &n);

		a = calloc(n, sizeof(int*));
		for (i=0; i<n; i++) {
			*(a+i) = calloc(n, sizeof(int));
			for (j=0; j<n; j++)
				*(*(a+i)+j) = rand()/((double)RAND_MAX +1) * 10;
		}

		for (i=0; i<n; printf("\n"), i++)
			for (j=0; j<n; j++)
				printf("%d ", * (*(a+i)+j));

		for (i=n-1; i>=0; i--)
			printf("%d ", *(*(a+i)+n-1-i));
		putchar('\n');

		return 0;
	}
